<script>
  const carousel = document.getElementById('testimonial-carousel');
  const slides = carousel.querySelectorAll('.carousel-track > div');
  const prevButton = document.getElementById('prev-button');
  const nextButton = document.getElementById('next-button');
  const currentSlideText = document.getElementById('current-slide');
  let currentSlideIndex = 0;

  function updateCarousel() {
    slides.forEach((slide, index) => {
      slide.classList.toggle('hidden', index !== currentSlideIndex);
    });
    currentSlideText.textContent = `${String(currentSlideIndex + 1).padStart(2, '0')}/${String(slides.length).padStart(2, '0')}`;
  }

  prevButton.addEventListener('click', () => {
    currentSlideIndex = (currentSlideIndex - 1 + slides.length) % slides.length;
    updateCarousel();
  });

  nextButton.addEventListener('click', () => {
    currentSlideIndex = (currentSlideIndex + 1) % slides.length;
    updateCarousel();
  });

  // Initialize carousel
  updateCarousel();
</script>
